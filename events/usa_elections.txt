# Election of 1936: Garner vs Alf Landon vs Long vs Other Guy
# Before ; Democrats : Garner, Republicans : Curtis
country_event = {
	id = usa_elections.1
	title = usa_elections.1.t
	desc = usa_elections.1.d
	picture = GFX_report_event_usa_election_generic
	fire_only_once = yes 
	trigger = {
		tag = USA
		date > 1936.11.1
		date < 1936.12.1
	}
	
	option = { 	#Democrats
		name = usa_elections.1.a
		ai_chance = {
			factor = 25
			modifier = {
				factor = 50
				is_historical_focus_on = yes
			}
		}	
		USA = {
		set_politics = { ruling_party = democratic elections_allowed = yes }	
		set_country_flag = garner_president_1
		}
		hidden_effect = {
		news_event = { days = 1 id = usa.10002 }
		}
	}

	option = { #Republicans
		name = usa_elections.1.b
		ai_chance = {
			factor = 25
			modifier = {
				factor = 50
				is_historical_focus_on = yes
			}
		}	
		USA = {
		set_country_flag = landon_president_1
		create_country_leader = {
	name = "Alf Landon"
	desc = "POLITICS_ALF_LANDON_DESC"
	picture = "alf_landon.dds"
	expire = "1965.1.1"
	ideology = conservatives_subtype
	traits = {
	staunch_constitutionalist
	}
	}	
		
		}
		hidden_effect = {
		news_event = { days = 1 id = usa.10002 }
		}
		}
	option = { #American First
		name = usa_elections.1.c
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}		
		USA = {
		set_country_flag = long_president_1
		set_politics = { ruling_party = neutrality elections_allowed = yes }
		}
		hidden_effect = {
		news_event = { days = 1 id = usa.10002 }
		#country_event = { days = 7 id = usa.200 }
		}
		}
	option = { #Socialist
		name = usa_elections.1.e
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}		
		USA = {
		set_country_flag = wheeler_president_1
		set_politics = { ruling_party = socialist elections_allowed = yes }
		}
		hidden_effect = {
		news_event = { days = 1 id = usa.10002 }
		#country_event = { days = 7 id = usa.10000 }
		}
		}		
}
