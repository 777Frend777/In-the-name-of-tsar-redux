on_actions = {
    on_puppet = { 
        effect = {
            if = {
                limit = {
                    original_tag = SLV
                    overlord = { original_tag = AUS }
                }
                set_cosmetic_tag = SLV_AUS
            }
            else_if = {
                limit = {
                    original_tag = SLV
                    overlord = {
                        NOT = {
                            original_tag = AUS
                        }
                    }
                }
                drop_cosmetic_tag = yes
            }
        }
    on_liberate = {
