RMN_japanese_fall = {
	if = {
		limit = { has_idea = RMN_japanese_down }
		modify_timed_idea = {
			idea = RMN_japanese_down
			days = 50
		}
	}
	else = {
		add_timed_idea = {
			idea = RMN_japanese_down
			days = 50
		}
	}
}

	}
	else = {
		add_timed_idea = {

		}
	}
}
