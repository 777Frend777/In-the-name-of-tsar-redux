scripted_gui = {

	#######################################################
	#	COUNTRY SETTINGS
	#######################################################
	
	DS_config_COUNTRY = {
		context_type = player_context

		window_name = "DS_config_COUNTRY_window"
		
		visible = {
			has_variable = DS_configuration_target
			
			has_country_flag = DS_config_COUNTRY
		}

		properties = {

			DS_config_COUNTRY_3_button = {
				image = "[DS_GetCountryReinforcementsSetting]"
			}

			DS_config_COUNTRY_6_button = {
				image = "[DS_GetCountryPlanningEntrenchmentSetting]"
			}
		}
		
		effects = {

			DS_config_COUNTRY_custom_bonus_button_click = {
				if = { limit = { NOT = { has_country_flag = DS_config_CUSTOM_BONUS } } 
				
					DS_get_custom_bonus_settings = yes
					set_country_flag = DS_config_CUSTOM_BONUS 
				}
				else = { 
					clr_country_flag = DS_config_CUSTOM_BONUS 
					DS_set_custom_bonus_settings = yes
				}
			}

			DS_config_COUNTRY_3_button_click = {
				var:DS_configuration_target = { 
					if = { limit = { has_global_flag = DS_dynamic_division_spawning }
						
						if = { limit = { NOT = { has_country_flag = DS_division_spawning } }
							
							if = { limit = { NOT = { has_country_flag = DS_disable_division_spawning } }
								set_country_flag = DS_disable_division_spawning

								else = { clr_country_flag = DS_disable_division_spawning }
							}
						}

						if = { limit = { has_country_flag = DS_division_spawning }
							
							if = { limit = { NOT = { has_country_flag = DS_disable_division_spawning } }
								set_country_flag = DS_disable_division_spawning
								clr_country_flag = DS_division_spawning

								else = { clr_country_flag = DS_disable_division_spawning }
							}
						}
					}
					else = {
						clr_country_flag = DS_disable_division_spawning 

						if = { limit = { has_country_flag = DS_division_spawning }

							clr_country_flag = DS_division_spawning
							
							else = { set_country_flag = DS_division_spawning }
						}
					}
				}
			}

			DS_config_COUNTRY_6_button_click = {
				var:DS_configuration_target = { 
					if = { limit = { has_global_flag = DS_dynamic_boost_planning }
						
						if = { limit = { NOT = { has_country_flag = DS_boost_planning } }
							
							if = { limit = { NOT = { has_country_flag = DS_disable_boost_planning } }
								set_country_flag = DS_disable_boost_planning

								else = { clr_country_flag = DS_disable_boost_planning }
							}
						}

						if = { limit = { has_country_flag = DS_boost_planning }
							
							if = { limit = { NOT = { has_country_flag = DS_disable_boost_planning } }
								set_country_flag = DS_disable_boost_planning
								clr_country_flag = DS_boost_planning

								else = { clr_country_flag = DS_disable_boost_planning }
							}
						}
					}
					else = {
						clr_country_flag = DS_disable_boost_planning 

						if = { limit = { has_country_flag = DS_boost_planning }

							clr_country_flag = DS_boost_planning
							
							else = { set_country_flag = DS_boost_planning }
						}
					}
				}
			}

			###########################################
			
			# EXIT
			DS_config_COUNTRY_exit_click = {
				clr_country_flag = DS_config_COUNTRY
			
				clr_country_flag = DS_config_ARMY

				if = { limit = { has_country_flag = DS_config_CUSTOM_BONUS } 
				
					clr_country_flag = DS_config_CUSTOM_BONUS 
					DS_set_custom_bonus_settings = yes
				}
				
				var:DS_configuration_target = {		
					clr_country_flag = DS_player_is_configuring_this_country

					clr_country_flag = DS_player_is_configuring_this_country_@ROOT
				}

				clear_variable = DS_configuration_target
			}
		}
	}
