	DS_config_DIFFICULTY = {
		context_type = player_context

		window_name = "DS_config_DIFFICULTY_window"

		visible = {

				image = "[DS_GetDynamicPlanningEntrenchmentSetting]"
			}
		}

		effects = {

			DS_config_DIFFICULTY_custom_bonus_button_click = {
				if = { limit = { NOT = { has_country_flag = DS_config_CUSTOM_BONUS } } set_country_flag = DS_config_CUSTOM_BONUS }
				else = { clr_country_flag = DS_config_CUSTOM_BONUS }
			}

			DS_config_DIFFICULTY_3_button_click = {
				if = { limit = { has_global_flag = DS_dynamic_division_spawning }
					clr_global_flag = DS_dynamic_division_spawning
				}
				else = {
					set_global_flag = DS_dynamic_division_spawning
				}
			}

			DS_config_DIFFICULTY_5_button_click = {
				if = { limit = { has_global_flag = DS_free_convoys }

				}
				else = {
					set_global_flag = DS_free_convoys
				}
			}

			DS_config_DIFFICULTY_6_button_click = {
				if = { limit = { has_global_flag = DS_dynamic_boost_planning }
					clr_global_flag = DS_dynamic_boost_planning
				}
				else = {
					set_global_flag = DS_dynamic_boost_planning
				}
			}

			###########################################
		
			DS_config_DIFFICULTY_button1_click = {
			
			}
			DS_config_DIFFICULTY_button2_click = {
				clr_country_flag = DS_config_DIFFICULTY
				clr_country_flag = DS_config_CUSTOM_BONUS
				set_country_flag = DS_config_GENERAL
			}

			DS_config_DIFFUCULTY_help_button_click = {
				if = { limit = { NOT = { has_variable = DS_config_DIFFICULTY_help_page } } set_variable = { DS_config_DIFFICULTY_help_page = 2 } }
				else_if = { limit = { check_variable = { DS_config_DIFFICULTY_help_page = 2 } } set_variable = { DS_config_DIFFICULTY_help_page = 3 } }
				else_if = { limit = { check_variable = { DS_config_DIFFICULTY_help_page = 3 } } clear_variable = DS_config_DIFFICULTY_help_page }
			}

			# EXIT
			DS_config_exit_click = {
				clr_country_flag = DS_config_DIFFICULTY
				
				clr_country_flag = DS_config_GENERAL

				clr_country_flag = DS_config_CUSTOM_BONUS
				
				clr_country_flag = DS_show_country_config_decisions
			}
		}
	}
